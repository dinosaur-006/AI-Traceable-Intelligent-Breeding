// 核心：模拟流式输出与侧边栏联动
function generateAIResponse(userText) {
    let responseText = "";
    let matchedKeyword = null;

    // 简单的关键词匹配逻辑
    if (userText.includes("熬夜")) {
        responseText = "针对您提到的**熬夜**情况，中医认为这容易耗伤心血和肾阴。建议您尝试简单的食补方案：<br><br>1. **西洋参**枸杞茶：补气滋阴。<br>2. 保持子午觉（中午11-13点小憩）。<br><br>需要为您生成具体的食谱吗？";
        matchedKeyword = "熬夜";
    } else if (userText.includes("气虚")) {
        responseText = "气虚体质的人就像电池没充满电。建议多吃黄色食物，如小米、土豆。药膳方面推荐**黄芪**炖鸡。";
        matchedKeyword = "气虚";
    } else {
        responseText = "收到了。根据您的描述，我正在检索相关的溯源滋补方案...（这是一个通用回复演示）";
    }

    // 触发流式打字效果
    const msgDiv = document.createElement('div');
    msgDiv.className = `message ai`;
    const bubble = document.createElement('div');
    bubble.className = "bubble";
    msgDiv.appendChild(bubble);
    chatWindow.appendChild(msgDiv);

    let i = 0;
    // 使用 HTML 解析稍显复杂，这里简化为纯文本打字，或者直接一次性渲染HTML
    // 为了效果好，我们直接渲染 HTML 但加一个淡入动画
    bubble.innerHTML = responseText;
    scrollToBottom();

    // 联动右侧面板
    if (matchedKeyword && knowledgeBase[matchedKeyword]) {
        updateKnowledgePanel(knowledgeBase[matchedKeyword]);
        
        // 连锁反应：如果主要建议里包含了具体的品类（如西洋参），也延时显示出来
        if (matchedKeyword === "熬夜") {
            setTimeout(() => {
                updateKnowledgePanel(knowledgeBase["西洋参"]);
            }, 1500);
        }
    }
}

function updateKnowledgePanel(data) {
    const div = document.createElement('div');
    div.className = "knowledge-card";
    div.style.animation = "fadeInUp 0.5s ease";
    
    let content = `<div class="k-title">${data.title}</div><div class="k-desc">${data.desc}</div>`;
    
    if (data.products) {
        content += `<div style="margin-top:10px; padding-top:10px; border-top:1px dashed #eee; font-size:0.8rem; color: var(--primary-color);">推荐关联：${data.products.join(', ')}</div>`;
    }
    
    if (data.isProduct) {
        content += `<div style="margin-top:10px; text-align:right;"><button style="background:var(--accent-color); color:white; border:none; padding:4px 12px; border-radius:4px; cursor:pointer;">查看溯源证书</button></div>`;
    }

    div.innerHTML = content;
    knowledgeContainer.prepend(div); // 最新知识插在最上面
}
